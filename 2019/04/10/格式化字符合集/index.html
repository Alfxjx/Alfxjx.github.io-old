连字符字符串转驼峰

border-bottom-color =>borderBottomColor

// solution1 按照分隔符拆开数组
function transform(str){
    var strArr = str.split('-');
    for(let i=1;i<strArr.length;i++){
        strArr[i] = strArr[i].charAt(0).toUpperCase()+strArr.substring(1);
    }
    return strArr.join('');
}
console.log(transform(('border-bottom-color')))

// solution2 拆成字符流来判断
function transform(str){
    let strArr = str.split('');
    for(let i=0;i<strArr.length;i++){
        if(strArr[i]==='-'){
            strArr.splice(i,1);
            if(i<strArr.length){
                strArr[i] = strArr[i].toUpperCase();
            }
        }
    }
    return strArr.join('');
}
console.log(transform(('border-bottom-color')))

// solution3 正则
function transform(str){
    const regex = /-(\w)/g;
    return str.replace(regex, function($0,$1){
        return $1.toUpperCase();
    })
}
console.log(transform(('border-bottom-color')))

‘\$0’代表完整的模式匹配文本，’\$1’代表的是第一个捕获组。

function transform(str){
    let temp = str.replace(/[A-Z]/g, function(match){
        return '-' + match.toLowerCase();
    });
    if(temp.slice(0,1)==='-'){
        temp = temp.slice(1);
    }
    return temp;
}

// reduce方法
function transform(prev,curr,index,array){
    if(/[A-Z]/.test(curr)){
        curr = curr.toLowerCase();
        if(index === 0){
            return prev+curr;
        } else {
            return prev+'-'+curr;
        }
    } else {
        return prev+curr;
    }
}
function toLower(arr){
    if(typeof arr === 'string'){
        arr = arr.split('');
    }
    return arr.reduce(transform, '');
}

// map
function transform(val,index,arr){
    if(/[A-Z]/.test(val)){
        if(index===0){
            return val.toLowerCase();
        } else {
            return '-'+val.toLowerCase();
        }
    } else {
        return val;
    }
}
function toLower(str){
    if(typeof str === 'string'){
        str = str.split('');
        return str.map(transform).join('');
    } else {
        return str.map(transform).join('');

// 不用正则，使用字符流的方法。
var cssString = ':root >div{\n\tbackground: red;\n\tfont-size: 17px;\n}';
function cssToReact(str){
    if(typeof str ==='string'){
        str = str.split('');
    }
    let arr = str;
    let start, end;
    for(let i=0;i<str.length;i++){
        if(str[i]==='{') start = i+2;
        if(str[i]==='}') end = i;
    }
    let temp = str.slice(start,end);
    for(let j=0;j<temp.length;j++){
        if(temp[j]==='-'){
            temp.splice(j,1);
            if(j<temp.length){
                temp[j] = temp[j].toUpperCase();
            }
        }
    }
    let ret = arr.slice(0,start).concat(temp,arr.slice(end,arr.length));
    return ret.join('');

}

金融数字格式化

要求： 数字保留两位小数。

123 => 123.00
123.456 => 123.46
-2.3333 => -2.33
123412 => 123,412

// 法1：toLocaleString
// 浏览器提供该 api 对数字进行格式化，不过对于小数的处理使用的是四舍五入
let num = 1234567;
console.log(num.toLocaleString('en-US')); // 1,234,567
num = 12345.6785;
console.log(num.toLocaleString('en-US')); // 12,345.679

// 法2：数组操作
function format(num) {
  if (typeof num !== 'number') return num;

  const [ first, last ] = ('' + num).split('.');
  const flen = first.length;
  const tmp = first.split('').reverse().reduce((res, n, i) => {
    if ((i + 1) % 3 === 0  && i !== flen - 1) res.push(n, ',');
    else res.push(n);
    return res;
  }, []).reverse().join('');

  return last ? tmp + '.' + last : tmp;
}

console.log(format(1234567)); // 1,234,567
console.log(format(12345.6785)); // 12,345.6785
console.log(format(123456.78)); // 123,456.78

格式化发布时间

function format(date, testMode) {
  const MIN = 60 * 1000;
  const HOUR = 60 * MIN;
  const DAY = HOUR * 24;
  const WEEK = DAY * 7;
  const time = testMode ? date : Date.now() - date;

  if (time < MIN) return '刚刚';

  if (time < HOUR) return Math.floor(time / MIN) + '分前';

  if (time < DAY) return Math.floor(time / HOUR) + '小时前';

  if (time < WEEK) return Math.floor(time / DAY) + '天前';

  return new Date(testMode ? Date.now() - date : date).toLocaleString();
}

// 测试
console.log(format(60 * 1000 - 1, true)); // 59 秒 999
console.log(format(60 * 1000, true)); // 1 分

console.log(format(60 * 1000 * 59 + 59 * 1000 + 999, true)); // 59 分 59 秒 999
console.log(format(60 * 1000 * 60, true)); // 1 小时

console.log(format(60 * 1000 * 60 * 23 + 60 * 1000 * 59 + 59 * 1000 + 999, true)); // 23 小时 59 分 59 秒 999
console.log(format(60 * 1000 * 60 * 24, true)); // 1 天

console.log(format(60 * 1000 * 60 * 24 * 6, true)); // 6 天
console.log(format(60 * 1000 * 60 * 24 * 7, true)); // 7 天

// 正常使用
console.log(format(1554111847534)); // 发布时的时间戳
