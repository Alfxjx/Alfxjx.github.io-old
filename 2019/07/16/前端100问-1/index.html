第 67 题 数组分组

随机生成一个长度为 10 的整数类型的数组，例如 [2, 10, 3, 4, 5, 11, 10, 11, 20]，将其排列成一个新数组，要求新数组形式如下，例如 [[2, 3, 4, 5], [10, 11], [20]]。

这个题主要是做三件事，一个是排序，一个是去重，还有一个出现了一点点分歧，不过就是分组和连续数字两种。

const array = [2, 10, 3, 4, 5, 11, 10, 11, 20];

function transform(arr){
  arr.sort((a,b)=>{return a-b});
  let _arr = [];
  for(let i=0;i<arr.length;i++){
    if(_arr.indexOf(arr[i])<0){
      _arr.push(arr[i])
    }
  }
  let continueArr = [], tempArr = [];
  _arr.map((e, index) => {
      tempArr.push(e);
      if (_arr[index+1] !== ++e) {
          continueArr.push(tempArr);
          tempArr = [];
      }
  });
  return continueArr;
}

console.log(transform(array));

第 77 题：旋转数组算法题

给定一个数组，将数组中的元素向右移动 k 个位置，其中 k 是非负数。

输入: [1, 2, 3, 4, 5, 6, 7] 和 k = 3
输出: [5, 6, 7, 1, 2, 3, 4]

// 右移
const rotate1 = (arr, num) => {
    for (let i = 0; i < num; i++) {
        let val = arr.pop();
        arr.unshift(val);
    }
    return arr;
};

// 左移
const rotate2 = (arr, num) => {
    for (let i = 0; i < num; i++) {
    let val = arr[0];
    arr.shift();
        arr.push(val);
    }
    return arr;
};

console.log(rotate1([-1, -100, 3, 99], 1));
console.log(rotate2([-1, -100, 3, 99], 1));

第 81 题：打印出 1 - 10000 之间的所有对称数

打印出 1 - 10000 之间的所有对称数

例如：121、1331 等

const symmetry = (startIndex, endIndex) => {
    let res = [];
    for (let i = startIndex; i < endIndex; i++) {
        let temp = (i).toString().split('');
        if (temp.join() == temp.reverse().join()) {
            res.push(i);
        }
    }
    return res;
};

console.log(symmetry(1, 200));
